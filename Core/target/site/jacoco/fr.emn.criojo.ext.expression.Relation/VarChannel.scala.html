/** Channel Variable */
case class VarChannel(n: String) extends Var[ChannelLocation](n)
with Relation {
}

object VarChannel {
  private var instanceNum = 0

  def apply() = new VarChannel("VarChannel@" + getInstanceNum)

  private def getInstanceNum = {
    instanceNum += 1
    instanceNum
  }
}
