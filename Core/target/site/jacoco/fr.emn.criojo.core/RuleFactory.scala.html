/**
 * The RuleFactory trait
 * @define THIS RuleFactory
 */
trait RuleFactory{
  type Body = List[Atom]
  type Head = List[Atom]

  def createRule(h:Head,b:Body,g:Guard):Rule = createRule(h,b,g,Set())
  def createRule(h:Head,b:Body,g:Guard,scope:Set[Variable]):Rule

}
